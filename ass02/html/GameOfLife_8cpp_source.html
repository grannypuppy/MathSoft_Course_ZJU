129void GameOfLife::update() {
130 // Calculate the next generation
131 for (int i = 0; i < height_; i++) {
132 for (int j = 0; j < width_; j++) {
133 int neighbors = countNeighbors(i, j);
134
135 // Apply Conway's Game of Life rules
136 if (grid_[i][j]) {
137 // Cell is alive
138 if (neighbors < 2 || neighbors > 3) {
139 nextGrid_[i][j] = false; // Cell dies
140 } else {
141 nextGrid_[i][j] = true; // Cell survives
142 }
143 } else {
144 // Cell is dead
145 if (neighbors == 3) {
146 nextGrid_[i][j] = true; // Cell becomes alive
147 } else {
148 nextGrid_[i][j] = false; // Cell stays dead
149 }
150 }
151 }
152 }
153
154 // Update the grid with the new generation
155 grid_ = nextGrid_;

159 int count = 0;
160
161 // Check all 8 neighboring cells
162 for (int i = -1; i <= 1; i++) {
163 for (int j = -1; j <= 1; j++) {
164 // Skip the cell itself
165 if (i == 0 && j == 0) continue;
166
167 // Calculate neighbor coordinates with wrapping
168 int neighborRow = (row + i + height_) % height_;
169 int neighborCol = (col + j + width_) % width_;
170
171 // Count live neighbors
172 if (grid_[neighborRow][neighborCol]) {
173 count++;

175 }
